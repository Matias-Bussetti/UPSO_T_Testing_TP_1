  <script>
    // https://es.stackoverflow.com/questions/345142/c%C3%B3mo-desabilitar-un-input-con-javascript
    // https://es.stackoverflow.com/questions/404858/habilitar-input-desde-javascript
    /**
     * Functions
     *
     */

    function $(element_id) {
      return document.getElementById(element_id);
    }
    function swapInputDisabled(element_id) {
      const state = $(element_id).disabled;
      $(element_id).disabled = !state;
    }
    function disableInput(element_id) {
      $(element_id).disabled = false;
    }
    /**
     * Event listeners
     *
     */
    $('edith').addEventListener("click", () => {
            disableInput('profe');
            disableInput('date');
            disableInput('hour');
    });
    $('save').addEventListener("click", () => {
            swapInputDisabled('profe');
            swapInputDisabled('date');
            swapInputDisabled('hour');
    })
  </script>
